  • Percentage = Sum / TotalMarks * 100

Code


 #include<stdio.h>

 int findSum(int lmarks1, int lmarks2, int lmarks3, int lmarks4, int lmarks5) {
  int sum;
  sum = lmarks1 + lmarks2 + lmarks3 + lmarks4 + lmarks5;
  return sum;
 }

 float findPercentage(int obtained, int total) {
  float percent;
  percent = obtained*100/(float)total;
  return percent;
 }

 int main() {
  int marks1, marks2, marks3, marks4, marks5;
  int sum, maxMarks;
  float percent;

  printf("Enter maxMarks: ");
  scanf("%d", &maxMarks);

  printf("Enter marks1: ");
  scanf("%d", &marks1);

  printf("Enter marks2: ");
  scanf("%d", &marks2);

  printf("Enter marks3: ");
  scanf("%d", &marks3);

  printf("Enter marks4: ");
  scanf("%d", &marks4);

  printf("Enter marks5: ");
  scanf("%d", &marks5);

  sum = findSum(marks1, marks2, marks3, marks4, marks5);
  percent = findPercentage(sum, 500);

  printf("Sum of all Marks: %d\n", sum);
  printf("Percentage gain: %.2f%%\n", percent);
  return(0);
 }
